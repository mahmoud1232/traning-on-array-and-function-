int max (int b[])
{
	
 	int max=b[0] ;

int sum (int b[] )
{

int s= 0 ; 
for(int i=0;i<size ; i++)
    s+=b[i];

}
void sort (int b [])
{
	int temp ; 
	for(int i=0;i<size;i++)
	for(int j=0;j<size;j++)
	if (b[i]<b[j])
	{
  		temp = b[i] ; 
  		b[i]=b[j];
 		b[j]=temp ; 
	  	
    }
	  
} 
int avg(int b[], int length_array){
